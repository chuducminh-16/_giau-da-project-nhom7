            stage.setTitle(title);

    public void connect(String host, int port) throws IOException {

        out = new PrintWriter(socket.getOutputStream(), true);
        in  = new BufferedReader(new InputStreamReader(socket.getInputStream()));

        // Thread ná»n: liÃªn tá»¥c Ä‘á»c message tá»« server

import com.auction.client.network.Message;

    private static final int PORT = 9090;

    // Thread-safe set chá»©a táº¥t cáº£ client Ä‘ang káº¿t ná»‘i
    private final Set<ClientHandler> clients =

    // Thread pool â€” má»—i client 1 thread
    private final ExecutorService pool =
            Executors.newCachedThreadPool();

    public void start() {
        System.out.println("[Server] Äang khá»Ÿi Ä‘á»™ng trÃªn port " + PORT);
        try (ServerSocket serverSocket = new ServerSocket(PORT)) {

                System.out.println("[Server] Client má»›i: "

                        new ClientHandler(socket, this);
                clients.add(handler);
                pool.submit(handler);
            }

            System.err.println("[Server] Lá»—i: " + e.getMessage());
        }

import java.util.List;
import java.util.concurrent.CopyOnWriteArrayList;

/**
 * Subject (Observable) â€” quáº£n lÃ½ danh sÃ¡ch observer vÃ  broadcast sá»± kiá»‡n.
 *
 * DÃ¹ng CopyOnWriteArrayList: an toÃ n khi nhiá»u thread cÃ¹ng Ä‘á»c/ghi.
 * AuctionManager extend class nÃ y Ä‘á»ƒ cÃ³ kháº£ nÄƒng notify.
 */
public abstract class AuctionSubject {

    // Thread-safe list: subscribe/unsubscribe Ã­t, notify nhiá»u â†’ phÃ¹ há»£p COWAL
    private final List<AuctionObserver> observers = new CopyOnWriteArrayList<>();

    // â”€â”€ Subscribe / Unsubscribe â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€

    public void subscribe(AuctionObserver observer) {
        if (observer != null && !observers.contains(observer)) {
            observers.add(observer);
        }
    }

    public void unsubscribe(AuctionObserver observer) {
        observers.remove(observer);
    }

    // â”€â”€ Notify helpers (gá»i tá»« subclass) â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€

     */
    protected void notifyBidPlaced(BidEvent event) {
        for (AuctionObserver obs : observers) {

                obs.onBidPlaced(event);
            } catch (Exception e) {
                System.err.println("[AuctionSubject] Observer lá»—i khi nháº­n bid: " + e.getMessage());
            }
        }
    }

    protected void notifyAuctionClosed(long auctionId, String winnerId, double finalPrice) {
        for (AuctionObserver obs : observers) {

            } catch (Exception e) {

            }
        }
    }

    public int getObserverCount() { return observers.size(); }

    /**
     * MÃ´ phá»ng AuctionService.placeBid() thread-safe vá»›i ReentrantLock.
     * DÃ¹ng Ä‘á»ƒ kiá»ƒm tra khÃ´ng cÃ³ lost update khi nhiá»u thread Ä‘áº·t giÃ¡.
     */
    static class SafeBidProcessor {
        private double currentPrice;
        private final java.util.concurrent.locks.ReentrantLock lock =
                new java.util.concurrent.locks.ReentrantLock();
        private final List<Double> acceptedBids = new CopyOnWriteArrayList<>();

        SafeBidProcessor(double startPrice) {
            this.currentPrice = startPrice;
        }

        BidResult bid(double amount) {
            lock.lock();
            try {
                if (amount <= currentPrice) return BidResult.PRICE_TOO_LOW;
                currentPrice = amount;
                acceptedBids.add(amount);
                return BidResult.SUCCESS;
            } finally {
                lock.unlock();
            }
        }

        double getCurrentPrice() { return currentPrice; }
        List<Double> getAcceptedBids() { return acceptedBids; }
    }

    void concurrentBidding_noLostUpdate() throws InterruptedException {
        SafeBidProcessor processor = new SafeBidProcessor(1000.0);
        int threadCount = 50;
        ExecutorService pool = Executors.newFixedThreadPool(threadCount);
        CountDownLatch latch = new CountDownLatch(threadCount);

        // Má»—i thread Ä‘áº·t giÃ¡ báº±ng index Ã— 100 (1100, 1200, ..., 6000)
        for (int i = 1; i <= threadCount; i++) {
            final double bid = 1000.0 + i * 100.0;
            pool.submit(() -> {
                processor.bid(bid);
                latch.countDown();
            });
        }

        latch.await(5, TimeUnit.SECONDS);
        pool.shutdown();

    }
